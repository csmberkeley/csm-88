\question What would Python display? If a StopIteration Exception
occurs, write \texttt{StopIteration}, or if another error occurs,
write \texttt{Error}. 

\begin{lstlisting}
>>> def weird_gen(x):
... 	if x % 2 == 0:
... 	   yield x * 2
... 	else:
...        yield x
... 	   yield from weird_gen(x - 1)
>>> next(weird_gen(2))
\end{lstlisting}
\begin{solution}[.25in]
\begin{lstlisting}
4
\end{lstlisting}
\end{solution}

\begin{lstlisting}
>>> list(weird_gen(3))
\end{lstlisting}
\begin{solution}[.25in]
\begin{lstlisting}
[3, 4]
\end{lstlisting}
\end{solution}
