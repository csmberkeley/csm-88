\question Write a function that returns the number of branches of a tree.

\begin{lstlisting}
def num_branches(t)
    """Return the number of branches of a tree.

    >>> t = tree(1, [tree(2), tree(3)])
    >>> num_branches(t)
    2
    """
\end{lstlisting}
\begin{solution}[.5in]
\begin{lstlisting}
    return len(branches(t))
\end{lstlisting}
\end{solution}
