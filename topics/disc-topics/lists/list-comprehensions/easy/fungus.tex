\question What would Python print?
\begin{lstlisting}
>>> lst = [3, 2, 1, 0]
>>> def fungus(spore):
        x = 0
        while spore[x] != 0:
            print('Mushroom!')
            x += 1
        return x

>>> fungus(lst)
\end{lstlisting}

\begin{solution}[3cm]
\begin{lstlisting}
Mushroom!
Mushroom!
Mushroom!
3
\end{lstlisting}
\end{solution}
