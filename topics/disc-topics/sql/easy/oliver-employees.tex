\question Write a query that outputs the names of employees that Oliver
Warbucks directly supervises.
\begin{solution}[1cm]
\begin{lstlisting}[language=SQL]
SELECT name FROM records WHERE supervisor = "Oliver Warbucks";
\end{lstlisting}
