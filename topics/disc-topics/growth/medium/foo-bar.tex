\begin{blocksection}
\question \begin{lstlisting}
def bar(n):
    if n % 2 == 1:
        return n + 1
    return n

def foo(n):
    if n < 1:
        return 2
    if n % 2 == 0:
        return foo(n - 1) + foo(n - 2)
    else:
        return 1 + foo(n - 2)
\end{lstlisting}

What is the order of growth of {\tt foo(bar(n))}?

\begin{solution}[0.0in]
    $\growth(n^2)$
\end{solution}
\end{blocksection}
