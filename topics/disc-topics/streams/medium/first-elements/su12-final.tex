\question (Summer 2012 Final) What are the first five values in the following
stream?

\begin{lstlisting}
def my_stream():
    def compute_rest():
        return add_streams(filter_stream(lambda x: x % 2 == 0, my_stream()),
                           map_stream(lambda x: x + 2, my_stream()))
    return Stream(2, compute_rest)
\end{lstlisting}
\begin{solution}[1cm]
2, 6, 14, 30, 62
\end{solution}
