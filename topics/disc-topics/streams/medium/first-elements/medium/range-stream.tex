\question Write a function \texttt{range-stream} which takes a \texttt{start} and
\texttt{end} argument, and returns a stream that represents the integers between
\and included \texttt{start} and \texttt{end - 1}.

\begin{lstlisting}
(define (range-stream start end)
\end{lstlisting}
\begin{solution}[1in]
\begin{lstlisting}
  (if (eq? start end)
    nil
    (cons-stream start (range-stream (+ start 1) end))
  )
)
\end{lstlisting}
\end{solution}
