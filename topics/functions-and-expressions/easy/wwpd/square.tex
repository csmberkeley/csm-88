\question Explain the difference between the following:


\begin{lstlisting}
>>> def square(x):
...     return x * x
\end{lstlisting}

\begin{solution}
Defines a function called \lstinline$square$.
\end{solution}

\lstinline$>>> square(4)$

\begin{solution}
Calls the function, \lstinline$square$.
\end{solution}

\lstinline$>>> square$

\begin{solution}
Evaluates to the function that squares its input number.
\end{solution}
