\question Write a function that returns the sum of all even numbers from 2 to n. 

(Hint: If n = 10, return 2 + 4 + 6 + 8 + 10)

Assume n is always greater than or equal to 2.

\begin{lstlisting}
def sum_even_to(n): 
    """
    >>> sum_even_to(6) # 2 + 4 + 6
    12
    >>> sum_even_to(10) # 2 + 4 + 6 + 8 + 10
    30
    >>> sum_even_to(11) # Still 2 + 4 + 6 + 8 + 10
    30
    """

\end{lstlisting}

\begin{solution}[1.5in]
\begin{lstlisting}
# For loop solution
def sum_even_to(n): 
    total = 0
    for i in range(2, n+1):
        if i % 2 == 0:
            total = total + i
    return total

# While loop solution
def sum_even_to(n): 
    total = 0
    while n > 0:
        if n % 2 == 0:
            total = total + n
        n = n - 1
    return total

# Another while loop solution
def sum_even_to(n):
    total = 0
    i = 2
    while i <= n:
        total = total + i
        i = i + 2
    return total
\end{lstlisting}
\end{solution}
