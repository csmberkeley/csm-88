class Tree:
    def __init__(self, label, branches=[]):
        for b in branches:
            assert isinstance(b, Tree)
        self.label = label
        self.branches = list(branches)

    def is_leaf(self):
        return not self.branches

    def mystery1(self):
        s = self.label
        for b in self.branches:
            s += b.label
        return s

    def mystery2(self):
        s = self.label
        for b in self.branches:
            s += b.mystery2()
        return s

    def max_label(self):
        answer = self.label
        for b in self.branches:
            x = b.max_label()
            if x > answer:
                answer = x
        return answer

    def mystery3(self):
        return self.label > max([b.max_label() for b in self.branches])
